1、纯粹的函数调用(全局性调用)，this代表global

    var x = 1;
    function test(){
        alert(this.x = 2);
        var a = this;
        alert(a); //[object Window]
    }
    test(); // 1

2、作为对象方法的调用，this就指这个上级对象。

    function test(){
        alert(this.x);
        var a = this;
        alert(a);//[object object]
    }
    var o = {};
    o.x = 1;
    o.m = test;
    o.m(); // 1

3.作为构造函数调用,this就指这个新对象。

    var x = 2;
    function test(){
        this.x = 1;
    }
    var o = new test();
    alert(o.x); //1;
    alert(x); //2
